function useCounter({value}) {
    const [count, setCount] = useState(value)
    const increment = () => setCount(value + 1)
    return {count, increment}
}

function Example() {
  const counter = userCounter({value: 0})
  return <div>
    <p>{counter.count}</p>
    <button onClick={counter.increment}>+</button>
  </div>
}
